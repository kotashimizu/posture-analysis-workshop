```ts
export function captureVideoFrame(video: HTMLVideoElement): { canvas: HTMLCanvasElement; previewUrl: string } {
  const canvas = document.createElement("canvas");
  canvas.width = video.videoWidth;
  canvas.height = video.videoHeight;
  const ctx = canvas.getContext("2d");
  if (!ctx) throw new Error("Canvas context が取れません");
  ctx.drawImage(video, 0, 0, canvas.width, canvas.height);
  return { canvas, previewUrl: canvas.toDataURL("image/png") };
}

export function seekVideo(video: HTMLVideoElement, time: number): Promise<void> {
  return new Promise<void>((resolve, reject) => {
    const duration = Number.isFinite(video.duration) ? video.duration : 0;
    const targetTime = Math.min(Math.max(time, 0), duration);
    if (Math.abs(video.currentTime - targetTime) < 0.01) {
      resolve();
      return;
    }
    const handler = () => {
      video.removeEventListener("seeked", handler);
      video.removeEventListener("error", errorHandler);
      resolve();
    };
    const errorHandler = (e: Event) => {
      video.removeEventListener("seeked", handler);
      video.removeEventListener("error", errorHandler);
      reject(e);
    };
    video.addEventListener("seeked", handler);
    video.addEventListener("error", errorHandler);
    video.currentTime = targetTime;
  });
}
```

# 2. ベストフレーム検索

`src/lib/best-frame.ts` を作成：

```ts
import type { Landmark } from "@/types";

export type BestFrameResult = {
  timeSec: number;
  movementScore: number;
  totalSampled: number;
};

export type DetectAtTimeFn = (
  video: HTMLVideoElement,
  timeSec: number
) => Promise<Landmark[] | null>;

// TODO: 要臨床調整（仮値）
export async function findBestFrame(
  video: HTMLVideoElement,
  detectAtTime: DetectAtTimeFn,
  options?: {
    sampleIntervalSec?: number;
    maxSamples?: number;
    onProgress?: (current: number, total: number) => void;
    signal?: AbortSignal;
  }
): Promise<BestFrameResult | null> {
  const interval = options?.sampleIntervalSec ?? 0.4;
  const maxSamples = options?.maxSamples ?? 60;
  const duration = isFinite(video.duration) ? video.duration : 0;
  if (duration <= 0) return null;

  const sampleCount = Math.min(maxSamples, Math.max(3, Math.floor(duration / interval)));
  const actualInterval = duration / sampleCount;

  const samples: { time: number; landmarks: Landmark[] }[] = [];
  for (let i = 0; i < sampleCount; i++) {
    if (options?.signal?.aborted) return null;
    const t = i * actualInterval;
    options?.onProgress?.(i, sampleCount);
    const landmarks = await detectAtTime(video, t);
    if (landmarks && landmarks.length > 0) {
      samples.push({ time: t, landmarks });
    }
  }
  options?.onProgress?.(sampleCount, sampleCount);

  if (samples.length < 3) return null;

  let bestIndex = 1;
  let minMovement = Infinity;
  for (let i = 1; i < samples.length - 1; i++) {
    const movement =
      computeMovement(samples[i - 1].landmarks, samples[i].landmarks) +
      computeMovement(samples[i].landmarks, samples[i + 1].landmarks);
    if (movement < minMovement) {
      minMovement = movement;
      bestIndex = i;
    }
  }

  return {
    timeSec: samples[bestIndex].time,
    movementScore: minMovement,
    totalSampled: samples.length,
  };
}

function computeMovement(a: Landmark[], b: Landmark[]): number {
  let sum = 0;
  const len = Math.min(a.length, b.length);
  for (let i = 0; i < len; i++) {
    const dx = (a[i]?.x ?? 0) - (b[i]?.x ?? 0);
    const dy = (a[i]?.y ?? 0) - (b[i]?.y ?? 0);
    sum += Math.sqrt(dx * dx + dy * dy);
  }
  return len > 0 ? sum / len : Infinity;
}
```

# 3. VideoScrubber コンポーネント

`src/components/VideoScrubber.tsx` を作成：

- Props: video: VideoSource | null、onFrameSelected: (canvas, previewUrl) => void、onError: (msg) => void
- 内部State: currentTime、duration、analyzing、progress（{current, total}）
- video要素は非表示（display:none）でいいが、メタデータ取得用に存在させる
- スライダー（src/components/ui/slider.tsx）で時刻を選択
- 「このフレームで評価」ボタン：seekVideo して captureVideoFrame、onFrameSelected を呼ぶ
- 「ベストフレーム自動選択」ボタン（lucide-react の Wand2 or Sparkles アイコン）：
  - findBestFrame を実行（detectAtTime は内部で seekVideo + captureVideoFrame + pose.detect）
  - 進捗を Progress バーで表示
  - 完了したら seek してから onFrameSelected
  - AbortController で「中止」ボタンも置く
- video が null なら何も表示しない（return null）
